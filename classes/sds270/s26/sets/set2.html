Q4

subgame <- function(color, number, wager){
  if (color == "red"){
    win = wager
  } else {
    if (number >= 2 & number <= 10){
      win = -wager
    } else {
      win = 0
    }
  }
  return(win)
}

game <- function(money, ...){
  spin <- sample(1:3, size = 1, prob = c(0.2, 0.7, 0.1))
  if (spin == 1){
    money = money + 5
  } else if (spin == 2){
    color = sample(c("red", "black"), 1)
    number = sample(1:13, 1)
    money = money + subgame(color, number, ...)
  } else {
    money = 0
  }
  return(money)
}
game(20, wager = 10)
spin <- sample(1:3, size = 1, prob = c(0.5, 0.4, 0.1))
replicate(10, game(20, wager = 10))
